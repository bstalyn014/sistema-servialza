// Helpers
function construirCuadrilla(data) {
    if (data.cuadrilla) return data.cuadrilla;
    if (data.supervisor && data.obrero) return `${data.supervisor} / ${data.obrero}`;
    return '';
}
function separarCuadrilla(cuadrillaStr) {
    let supervisor = '', obrero = '';
    if (cuadrillaStr && cuadrillaStr.includes(' / ')) {
        const partes = cuadrillaStr.split(' / ');
        supervisor = partes[0] || '';
        obrero = partes[1] || '';
    }
    return { supervisor, obrero };
}
function obtenerPrecio(nombreItem) {
    const item = itemsCobroData.find(i => i.nombre === nombreItem);
    return item ? item.precio : 0;
}

function obtenerDatosMapa() {
  try {
    // Usar el spreadsheet dedicado exclusivamente al mapa
    const ss = SpreadsheetApp.openById(MAPA_SHEET_ID);
    const allSheets = ss.getSheets();
    
    // Filtrar solo hojas que empiecen con "Ordenes_Mapa"
    const mapSheets = allSheets.filter(s => s.getName().startsWith('Ordenes_Mapa'));
    
    if (mapSheets.length === 0) {
      return crearRespuestaJSON({ success: false, error: "No se encontraron hojas que empiecen con 'Ordenes_Mapa' en el spreadsheet del mapa." });
    }
    
    const resultados = [];
    
    for (const sheet of mapSheets) {
      const data = sheet.getDataRange().getValues();
      if (data.length <= 1) continue; // Hoja vacía o solo con encabezados
      
      // Cada hoja tiene sus propios headers (por si el orden de columnas varía)
      const localHeaders = data[0].map(h => h.toString().trim());
      const localXIndex = localHeaders.findIndex(h => h.toUpperCase() === 'X');
      const localYIndex = localHeaders.findIndex(h => h.toUpperCase() === 'Y');
      
      // Si esta hoja no tiene columnas X/Y, la saltamos sin error
      if (localXIndex === -1 || localYIndex === -1) continue;
      
      for (let i = 1; i < data.length; i++) {
        const row = data[i];
        // En Ecuador, X (Longitud) es aprox -79/-80 y Y (Latitud) es aprox -2. Leaflet usa [Lat, Lng], por ende [Y, X]
        const posX = parseFloat(row[localXIndex]);
        const posY = parseFloat(row[localYIndex]);
        
        if (isNaN(posX) || isNaN(posY)) continue;
        
        const item = { x: posX, y: posY, info: {} };
        
        localHeaders.forEach((header, index) => {
          if (index !== localXIndex && index !== localYIndex && header !== '') {
            item.info[header] = row[index];
          }
        });
        resultados.push(item);
      }
    }
    
    return crearRespuestaJSON({ success: true, data: resultados });
    
  } catch (error) {
    return crearRespuestaJSON({ success: false, error: error.toString() });
  }
}
// 🔥 LÓGICA DE RENDIMIENTO UNIFICADA 🔥
function obtenerResumenPresupuestario() {
  try {
    const ss = SpreadsheetApp.openById(SHEET_ID);
    const hoy = new Date();
    hoy.setHours(0,0,0,0);
    
    const totalesPorCuadrilla = {};
    // Helper para procesar hojas
    const procesarHoja = (nombreHoja, idxCuadrilla, idxPrecio) => {
        const sheet = ss.getSheetByName(nombreHoja);
        if (!sheet) return;
        
        const data = sheet.getDataRange().getValues();
        // Empezar en i=1 (saltar encabezado)
        for (let i = 1; i < data.length; i++) {
          const row = data[i];
          const fechaRow = new Date(row[0]); // Asumimos Fecha siempre en col A (0)
          
          if (esMismoDia(fechaRow, hoy)) {
            const cuadrilla = row[idxCuadrilla];
            const precio = parseFloat(row[idxPrecio]) || 0;
            
            if (cuadrilla && cuadrilla !== '') {
              if (!totalesPorCuadrilla[cuadrilla]) {
                totalesPorCuadrilla[cuadrilla] = 0;
              }
              totalesPorCuadrilla[cuadrilla] += precio;
            }
          }
        }
    };
    // 1. Procesar Reconexion
    // indices (0-based): Cuadrilla=2 (C), Precio=25 (Z)
    procesarHoja('Reconexion', 2, 25);
    // 2. Procesar Mantenimiento
    // indices (0-based): Cuadrilla=2 (C), Precio=25 (Z)
    // Nota: Confirma los índices basados en el appendRow de Mantenimiento arriba
    // rowData Mantenimiento: [Fecha(0), Contrato(1), Cuadrilla(2)... Item(24), Precio(25), Obs(26)]
    procesarHoja('Mantenimiento', 2, 25);
    
    // Convertir a Array ordenado
    const resultado = Object.keys(totalesPorCuadrilla).map(key => ({
      cuadrilla: key,
      total: Number(totalesPorCuadrilla[key].toFixed(2))
    })).sort((a, b) => b.total - a.total); 
    
    return crearRespuestaJSON({ 
      success: true, 
      data: resultado, 
      fecha: hoy.toLocaleDateString() 
    });
    
  } catch (error) {
    return crearRespuestaJSON({ success: false, error: error.toString() });
  }
}
function esMismoDia(d1, d2) {
  return d1.getFullYear() === d2.getFullYear() &&
         d1.getMonth() === d2.getMonth() &&
         d1.getDate() === d2.getDate();
}
function crearRespuestaJSON(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
